from itertools import combinations
 
def solve():
    N = int(input())
    S = [list(map(int, input().split())) for _ in range(N)]
    ingredients = set(range(N))
    min_diff = float('inf')
 
    # 1. N개 중 N/2개를 뽑는 모든 조합 생성
    for team_a_members in combinations(ingredients, N // 2):
        team_a_members = set(team_a_members)
        team_b_members = ingredients - team_a_members
 
        # 2. 각 팀의 시너지 점수 계산
        synergy_a = 0
        for i, j in combinations(team_a_members, 2):
            synergy_a += S[i][j] + S[j][i]
 
        synergy_b = 0
        for i, j in combinations(team_b_members, 2):
            synergy_b += S[i][j] + S[j][i]
         
        # 3. 시너지 차이 계산 및 최소값 갱신
        diff = abs(synergy_a - synergy_b)
        if diff < min_diff:
            min_diff = diff
 
    return min_diff
 
T = int(input())
for test_case in range(1, T + 1):
     result = solve()
     print(f"#{test_case} {result}")
